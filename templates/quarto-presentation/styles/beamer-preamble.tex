% --- Theme Configuration ---
\setbeamertemplate{navigation symbols}{}
\usefonttheme{professionalfonts}
\setbeamerfont{title}{size=\Large, series=\bfseries}
\setbeamerfont{frametitle}{size=\large, series=\bfseries}
\setbeamerfont{footnote}{size=\tiny}
\setbeamerfont{footline}{size=\tiny}

% --- Beamer Colors ---
\setbeamercolor{title}{fg=PrimaryColor, bg=white}
\setbeamercolor{frametitle}{fg=DarkGray, bg=white}
\setbeamercolor{structure}{fg=PrimaryColor}
\setbeamercolor{normal text}{fg=black!85}
\setbeamercolor{alerted text}{fg=PrimaryColor}
\setbeamercolor{footline}{fg=DarkGray!60, bg=white}

% --- Frame Title with Gray Underline ---
\setbeamertemplate{frametitle}{%
  \vskip6pt%
  \insertframetitle%
  \par\vskip-4pt%
  {\color{DarkGray}\rule{\textwidth}{1.5pt}}%
  \vskip4pt%
}

% --- Custom Footline (dynamic: reads from QMD metadata) ---
\setbeamertemplate{footline}{%
  \usebeamerfont{footline}%
  \usebeamercolor[fg]{footline}%
  \hspace{8pt}%
  \insertshortauthor~|~\insertshortinstitute~|~\insertframenumber/\inserttotalframenumber%
  \hspace{8pt}\vskip6pt%
}

% keybox --- key findings / takeaways
\newenvironment{keybox}{%
  \begin{tikzpicture}%
    \node[draw=PrimaryColor!70, line width=1pt, rounded corners=4pt,%
          fill=PrimaryColor!8, inner sep=8pt, text width=0.9\textwidth] \bgroup%
      {\small\color{PrimaryColor!70}\textbf{Key Point}}\par\vskip4pt%
      \color{black!85}%
}{%
    \egroup;%
  \end{tikzpicture}%
}

% quotebox --- block quotes with decorative left border
\newenvironment{quotebox}{%
  \begin{tikzpicture}%
    \node[draw=none, fill=SoftGray, inner sep=8pt,%
          text width=0.88\textwidth,%
          append after command={%
            (\tikzlastnode.north west) edge[line width=3pt, DarkGray!50]%
            (\tikzlastnode.south west)%
          }] \bgroup%
      \itshape\color{black!80}%
}{%
    \egroup;%
  \end{tikzpicture}%
}
